## Files

```
static-site-template/
  spin.toml

    package.json.tmpl
    src/
      index.html.tmpl
      main.js.tmpl
    posts/
      hello.md
    README.md.tmpl
```

## spin.toml

```toml
name = "static-site"
version = "0.1.0"
description = "Markdown static site starter"
type = "site"
language = "js"

[params]
site_name = { type = "text", prompt = "Site name", default = "My site" }
author = { type = "text", prompt = "Author name", default = "" }

[[post]]
run = "npm install"
```

## _base/package.json.tmpl

```json
{
  "name": "{{ .site_name | kebab }}",
  "version": "0.1.0",
  "scripts": {
    "dev": "vite",
    "build": "vite build"
  },
  "devDependencies": {
    "vite": "^6.0.0"
  }
}
```

## _base/src/index.html.tmpl

```html
<!doctype html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <title>{{ .site_name }}</title>
  </head>
  <body>
    <div id="app"></div>
    <script type="module" src="./main.js"></script>
  </body>
</html>
```

## _base/src/main.js.tmpl

```js
import postsGlob from '../posts/*.md'

const posts = Object.entries(postsGlob).map(([path, mod]) => ({
  path,
  ...mod.metadata,
  html: mod.html,
}));

document.querySelector('#app').innerHTML = `
  <h1>{{ .site_name }}</h1>
  <ul>
    ${posts.map(p => `<li><a href="#${p.path}">${p.title}</a></li>`).join('')}
  </ul>
`;
```

## Usage

```sh
spin new mysite --template ./static-site-template
```
