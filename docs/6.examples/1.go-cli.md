## Files

```
go-cli-template/

    cmd/
      root.go.tmpl

    README.md.tmpl
```

## spin.toml

```toml
name = "go-cli"
version = "0.1.0"
description = "Minimal Go CLI starter"
type = "cli"
language = "go"

[params]
project_name = { type = "text", prompt = "Project name", default = "mycli" }
module = { type = "text", prompt = "Go module path", default = "github.com/example/mycli" }
use_cobra = { type = "bool", prompt = "Use cobra?", default = true }

[[post]]
run = "go mod tidy"
```

## _base/go.mod.tmpl

```tmpl
module {{ .module }}

go 1.23

require (
{{ if .use_cobra }}    github.com/spf13/cobra v1.9.1
{{ end }}    charm.land/log/v2 v2.0.0
)
```

## _base/main.go.tmpl

```tmpl
package main

import (
{{ if .use_cobra }}    "{{ .module }}/cmd"
{{ else }}    "fmt"
{{ end }}
)

func main() {
{{ if .use_cobra }}    cmd.Execute()
{{ else }}    fmt.Println("Hello from {{ .project_name }}!")
{{ end }}
}
```

## _base/cmd/root.go.tmpl

```tmpl
package cmd

import (
    "fmt"
    "os"

    "github.com/spf13/cobra"
)

var rootCmd = &cobra.Command{
    Use:   "{{ .project_name | kebab }}",
    Short: "A brief description of {{ .project_name }}",
    Run: func(cmd *cobra.Command, args []string) {
        fmt.Println("Hello from {{ .project_name }}!")
    },
}

func Execute() {
    if err := rootCmd.Execute(); err != nil {
        os.Exit(1)
    }
}
```
