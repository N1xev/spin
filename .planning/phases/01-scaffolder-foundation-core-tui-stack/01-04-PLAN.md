    **scripts/check-v1-leaks.sh** -- Bash script (executable, `chmod +x`):
    ```bash
    #!/usr/bin/env bash
    # scripts/check-v1-leaks.sh
    # Greps a scaffolded project for v1 charmbracelet API leaks.
    # Usage: check-v1-leaks.sh <project-dir>
    # Exit 0 if clean, exit 1 if any v1 pattern is found.
    set -euo pipefail

    ROOT="${1:-./myapp}"
    if [[ ! -d "$ROOT" ]]; then
      echo "error: directory '$ROOT' does not exist" >&2
      exit 1
    fi

    # Go file patterns (PITFALLS #1, #2, #3, #4 from RESEARCH)
    declare -a GO_PATTERNS=(
      'github\.com/charmbracelet/'
      'View\(\) string'
      'tea\.WithAltScreen'
      'tea\.WithMouseCellMotion'
      'tea\.EnterAltScreen'
      'tea\.HideCursor'
      'tea\.ExitAltScreen'
      'lipgloss\.NewRenderer'
      'lipgloss\.DefaultRenderer'
      'lipgloss\.SetDefaultRenderer'
      'lipgloss\.AdaptiveColor\{'
      'lipgloss\.ColorProfile\('
      'lipgloss\.HasDarkBackground\(\)'
      'tea\.KeyCtrlC'
      'tea\.MouseButtonLeft'
      'tea\.MouseButtonRight'
      'tea\.MouseButtonMiddle'
      'msg\.Type'
      'msg\.Runes'
      'msg\.Alt'
      'msg\.X'
      'msg\.Y'
    )

    # .air.toml patterns (PITFALL #10)
    declare -a AIR_PATTERNS=(
      'bin\s*=\s*"tmp/main"'
    )

    FAIL=0
    for pat in "${GO_PATTERNS[@]}"; do
      if matches=$(grep -rEn --include='*.go' --include='*.tmpl' "$pat" "$ROOT" 2>/dev/null); then
        echo "FAIL: v1 pattern matched: $pat" >&2
        echo "$matches" >&2
        FAIL=1
      fi
    done

    AIR_FILE="$ROOT/.air.toml"
    if [[ -f "$AIR_FILE" ]]; then
      for pat in "${AIR_PATTERNS[@]}"; do
        if matches=$(grep -En "$pat" "$AIR_FILE" 2>/dev/null); then
          echo "FAIL: deprecated air pattern: $pat" >&2
          echo "$matches" >&2
          FAIL=1
        fi
      done
    fi

    if [[ $FAIL -ne 0 ]]; then
      exit 1
    fi
    echo "OK: no v1 leaks detected in $ROOT"
    ```
    Per RESEARCH §11.1, the patterns cover 22 Go patterns (slightly expanded from the 17 in RESEARCH.md) and 1 air pattern. The expansion to 22 covers the full set of v1->v2 removed APIs. After writing, `chmod +x scripts/check-v1-leaks.sh` so it can be run directly.
